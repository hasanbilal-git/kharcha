const SHEET = 'cost'; function doPost(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET); if (!sheet) sheet = ss.insertSheet(SHEET); if (sheet.getLastRow() === 0) { sheet.appendRow(['Date','Year','Month', 'Month (int)','Header','Type', 'Item','Amount','Account']); } const d = JSON.parse(e.postData.contents); sheet.appendRow([d.date,d.year,d.month, d.monthInt,d.header,d.type, d.item,d.amount,d.account]); return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ok:false,e:err.message})) .setMimeType(ContentService.MimeType.JSON); } } function doGet() { return ContentService .createTextOutput(JSON.stringify({ok:true,msg:'Kharcha connected'})) .setMimeType(ContentService.MimeType.JSON); }
