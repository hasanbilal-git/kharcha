const EXPENSES_SHEET = 'expenses'; const TRANSACTIONS_SHEET = 'transactions'; function doPost(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const d = JSON.parse(e.postData.contents); let ts = ss.getSheetByName(TRANSACTIONS_SHEET); if (!ts) ts = ss.insertSheet(TRANSACTIONS_SHEET); if (ts.getLastRow() === 0) { ts.appendRow(['Date','Year','Month','Month (int)', 'Entry Type','Header','Item','Amount', 'From Account','To Account']); } ts.appendRow([d.date,d.year,d.month,d.monthInt, d.entryType,d.header,d.item,d.amount, d.fromAccount,d.toAccount]); if (d.entryType === 'Expense') { // only expenses go to expenses sheet let es = ss.getSheetByName(EXPENSES_SHEET); if (!es) es = ss.insertSheet(EXPENSES_SHEET); if (es.getLastRow() === 0) { es.appendRow(['Date','Year','Month','Month (int)', 'Header','Type','Item','Amount','Account']); } es.appendRow([d.date,d.year,d.month,d.monthInt, d.header,d.type,d.item,d.amount,d.fromAccount]); } return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ok:false,e:err.message})) .setMimeType(ContentService.MimeType.JSON); } } function doGet() { return ContentService .createTextOutput(JSON.stringify({ok:true,msg:'Kharcha v2'})) .setMimeType(ContentService.MimeType.JSON); }
